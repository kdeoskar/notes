\underline{\textbf{Problem 1:}}

We have a wavefunction 
   \[ \psi(x) = \frac{A}{x^2 + a^2},\;\;for\;\;(-\infty < x < \infty) \]
   for positive constants $A$ and $a$.

\begin{enumerate}[(a)]
   \item First, we want to find $A$ by normalizing the wavefunction.
   
   We know 
   \[ \int_{\infty}^{\infty} dx\; |\psi(x)|^2 = 1 \]

   So, 
   \begin{align*}
      &\int_{\infty}^{\infty} dx\; |\psi(x)|^2 = 1 \\
      \implies&\int_{\infty}^{\infty} dx\; \psi(x)^{*}\psi(x) = 1 \\
      \implies&\int_{\infty}^{\infty} dx\; \left(\frac{A}{x^2 + a^2}\right)^{*} \left(\frac{A}{x^2 + a^2}\right) = 1 \\
      \implies&\int_{\infty}^{\infty} dx\; \left(\frac{A}{x^2 + a^2}\right)^{2} = 1 \\
      \implies&A^2 \cdot \int_{\infty}^{\infty} dx\; \left(\frac{1}{x^2 + a^2}\right)^2 = 1
   \end{align*}
